# Example: Next.js 15 MDX Blog with Velite

Full pipeline for a Next.js 15 App Router blog using Velite as the content compiler, with remark-gfm, remark-math, rehype-katex, and rehype-pretty-code (Shiki v4).

---

## Installation

```bash
npm i velite @next/mdx remark-gfm remark-math rehype-katex rehype-pretty-code
npm i -D @types/mdast @types/hast
```

## velite.config.ts

```typescript
import { defineConfig, s } from 'velite'
import remarkGfm from 'remark-gfm'
import remarkMath from 'remark-math'
import rehypeKatex from 'rehype-katex'
import { rehypePrettyCode } from 'rehype-pretty-code'
import rehypeSlug from 'rehype-slug'
import rehypeAutolinkHeadings from 'rehype-autolink-headings'

export default defineConfig({
  root: 'content',
  output: {
    data: '.velite',
    assets: 'public/static',
    base: '/static/',
    name: '[name]-[hash:6].[ext]',
    clean: true,
  },
  collections: {
    posts: {
      name: 'Post',
      pattern: 'posts/*.mdx',
      schema: s.object({
        title: s.string().max(99),
        date: s.isodate(),
        description: s.string().max(300).optional(),
        tags: s.array(s.string()).default([]),
        code: s.mdx(),
      }),
    },
  },
  mdx: {
    remarkPlugins: [remarkGfm, remarkMath],
    rehypePlugins: [
      rehypeKatex,
      [rehypePrettyCode, {
        theme: { dark: 'github-dark', light: 'github-light' },
      }],
      rehypeSlug,
      [rehypeAutolinkHeadings, { behavior: 'wrap' }],
    ],
  },
})
```

## next.config.mjs (Velite prebuild — Turbopack safe)

```mjs
// next.config.mjs
/** @type {import('next').NextConfig} */
const nextConfig = {
  pageExtensions: ['ts', 'tsx', 'md', 'mdx'],
}

// Velite prebuild: Turbopack-compatible approach
const isDev = process.argv.indexOf('dev') !== -1
const isBuild = process.argv.indexOf('build') !== -1
if (!process.env.VELITE_STARTED && (isDev || isBuild)) {
  process.env.VELITE_STARTED = '1'
  import('velite').then((m) => m.build({ watch: isDev, clean: !isDev }))
}

export default nextConfig
```

## mdx-components.tsx (required for App Router)

```tsx
// mdx-components.tsx (project root — required for @next/mdx App Router)
import type { MDXComponents } from 'mdx/types'
import Image from 'next/image'

export function useMDXComponents(components: MDXComponents): MDXComponents {
  return {
    // Override img to use next/image for optimization
    img: (props) => (
      <Image
        {...(props as React.ComponentProps<typeof Image>)}
        alt={props.alt ?? ''}
        sizes="(max-width: 768px) 100vw, 768px"
      />
    ),
    ...components,
  }
}
```

> The component map internals belong to `react-worker-bee`. This file is the handoff point between this stinger (pipeline config) and that worker-bee (component design).

## Blog page: app/blog/[slug]/page.tsx

```tsx
import { posts } from '.velite'  // generated by Velite
import { notFound } from 'next/navigation'
import { MDXContent } from '@/components/mdx-content'
import 'katex/dist/katex.min.css'    // required for math rendering
import '@/styles/code.css'           // rehype-pretty-code styles

interface Props {
  params: Promise<{ slug: string }>
}

export async function generateStaticParams() {
  return posts.map((post) => ({ slug: post.slug }))
}

export default async function BlogPost({ params }: Props) {
  const { slug } = await params
  const post = posts.find((p) => p.slug === slug)
  if (!post) notFound()

  return (
    <article>
      <h1>{post.title}</h1>
      <time dateTime={post.date}>{post.date}</time>
      <MDXContent code={post.code} />
    </article>
  )
}
```

## MDXContent component

```tsx
// src/components/mdx-content.tsx
'use client'
import * as runtime from 'react/jsx-runtime'
import { useMDXComponents } from '@/mdx-components'

interface Props {
  code: string
}

export function MDXContent({ code }: Props) {
  const components = useMDXComponents({})
  // Velite compiles MDX to a function body string; evaluate it here
  const fn = new Function('runtime', 'components', `
    const { jsx, jsxs, Fragment } = runtime
    ${code}
    return MDXContent({ components })
  `)
  return fn(runtime, components)
}
```
